前端组件组合架构

1. 组件独立性，扩展性，粘合性
2. 组件页面渲染——>DOM树数据结构
3. 页面数据渲染 ——>状态树数据结构
4. 页面事件冒泡 ——>修改状态数据
5. 数据渲染 ——> 依赖于状态树的改变

组件树
{
    "节点名": "",
    "位置": "",
    "属性": {
        "a": {
            "是否依赖状态树": true,
            "依赖状态树的路径": "page.data.a.active"
        }
    },
    "directives": {   // 指令，针对vue之类的
        "v-for": "(item,i) in currentList"
    },
    "事件": {
        "点击": "调用状态树set方法，调用事件树方法",
        "触碰"
    },
    "childrens": [
        "子节点"
    ]
}

状态树
{
    "page": {
        "data": {
            "a": "121",
        }
    }
}

事件树

    "事件名": ["http请求", "事件名， 事件颗粒化"],
    "事件名2": "自定义事件方法"
}
